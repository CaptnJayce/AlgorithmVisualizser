bogo sort

explanation

how does it work:

> bogo sort is the gambler's chosen sort
> it randomises the input set until it's sorted

use cases:

> there aren't any :(
> although can technically be the fastest sort for any use case.. if you're lucky enough

code:

the code is as follows: 

function bogoSort(array){
    var isSorted = function(array){
        for(var i = 1; i < array.length; i++){
            if (array[i-1] > array[i]) {
                return false;
            }
        }
        return true;
    };

    function shuffle(array){
        var count = array.length, temp, index;

        while(count > 0){
            index = Math.floor(Math.random() * count);
            count--;

            temp = array[count];
            array[count] = array[index];
            array[index] = temp;
        }
        return array;
    }

   function sort(array){
        var sorted = false;
        while(!sorted){
            array = shuffle(array);
            sorted = isSorted(array);
        }
        return array;
    }
    return sort(array);
}
