quick sort

explanation

how does it work:

> randomly select a 'pivot' point
> to one side of the pivot, sort values larger than the pivot, and lesser on the other side
> repeat until sorted

use cases:

> used in stock trading for financial analysis
> network routing
> recommendation systems, such as in ecommerce

function partition(array, low, high) { 
    let pivot = array[high]; 
    let i = low - 1;

    return i + 1; // Return the partition index 
} 
  
function quickSort(array, low, high) { 
    if (low >= high) return; 
    let pi = partition(array, low, high); 
  
    quickSort(array, low, pi - 1); 
    quickSort(array, pi + 1, high); 
}
